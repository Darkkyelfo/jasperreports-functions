# para usar no maven basta adicionar esse repositorio

  <repository>
	    <id>jasperreports-functions</id>
	    <name>jasperreports functions</name>
	    <url>https://github.com/Darkkyelfo/jasperreports-functions/raw/master/</url>
  </repository>
  
 #e adicionar essa dependencia
 <dependency>
    <groupId>jasperreports-functions</groupId>
    <artifactId>jasperreports-functions</artifactId>
    <version>6.4.3</version>
</dependency>
